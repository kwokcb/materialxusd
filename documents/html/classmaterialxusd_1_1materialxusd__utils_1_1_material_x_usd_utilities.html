36 """
37 @brief Encapsulate top level nodes in a nodegraph. Remap any connections to the top level nodes
38 to outputs on a new nodegraph.
39 @param input_path The path to the MaterialX document.
40 @param new_input_path The path to write the modified MaterialX document.
41 @param nodegraph_name The name of the new nodegraph to encapsulate the top level nodes. Default is 'top_level_nodes'.
42 @param remove_original_nodes If True, remove the original top level nodes from the document. Default is True.
43 @return The number of top level nodes found
44 """
45 connections_made = 0
46 top_level_nodes_found = 0
47
48 # Find all children of document which are no material or shader nodes.
49 top_level_nodes = []
50 top_level_connections = []
51 for elem in doc.getChildren():
52 # skips elements that are part of the stdlib
53 if elem.hasSourceUri():
54 continue
55 if (
56 elem.getName()
57 and (elem.getType() not in ["material", "surfaceshader"])
58 and elem.getCategory() not in ["nodegraph"]
59 ):
60 #print("Finding top level nodes: ", elem.getName(), elem.getType())
61 top_level_nodes.append(elem)
62 elif elem.getType() in ["surfaceshader"]:
63 for input_port in elem.getInputs():
64 upstream_node = input_port.getNodeName()
65 if len(upstream_node) > 0:
66 upstream_output_name = input_port.getOutputString()
67 #print("Store connection: ", upstream_node, "<--", input_port.getNamePath())
68 top_level_connections.append([upstream_node, input_port.getNamePath(), upstream_output_name])
69
70 #print("Top level connections: ", top_level_connections)
71 top_level_nodes_found = len(top_level_nodes)
72 if top_level_nodes_found == 0:
73 return top_level_nodes_found
74
75 # create nodegraph
76 ng_name = doc.createValidChildName(nodegraph_name)
77 ng = doc.addNodeGraph(ng_name)
78 for node in top_level_nodes:
79 #print("Adding node: ", node.getName())
80 new_node = ng.addNode(node.getCategory(), mx.createValidName(node.getName()), node.getType())
81 new_node.copyContentFrom(node)
82 for connect in top_level_connections:
83 if connect[0] == node.getName():
84 the_input = doc.getDescendant(connect[1])
85 if not the_input:
86 continue
87
88 # Create a new output on the graph
89 new_output = ng.addOutput(ng.createValidChildName("out"), the_input.getType())
90 new_output.setNodeName(new_node.getName())
91 if len(connect[2]) > 0:
92 new_output.setOutputString(connect[2])
93 #print("Create new output: ", mx.prettyPrint(new_output))
94 the_input.setNodeGraphString(ng_name)
95 the_input.setOutputString(new_output.getName())
96 the_input.removeAttribute("nodename")
97 #print(f"Reconnecting {the_input.getNamePath()} {connect[1]} to {mx.prettyPrint(the_input)}")
98 connections_made += 1
99
100 if remove_original:
101 for node in top_level_nodes:
102 doc.removeChild(node.getName())
103
104 return top_level_nodes_found
