18 def validate_stage(self, file:str, verboseOutput:bool=False):
19 '''
20 @brief This function validates a USD file using the ComplianceChecker.
21 @param file: The path to the USD file to validate.
22 @param verboseOutput: If True, the compliance check will output verbose information. Default is False.
23 @return: A tuple containing the errors, warnings, and failed checks.
24 '''
25 # Set up a ComplianceChecker
26 compliance_checker = UsdUtils.ComplianceChecker(
27 rootPackageOnly=False,
28 skipVariants=False,
29 verbose=verboseOutput
30 )
31
32 # Run the compliance check
33 compliance_checker.CheckCompliance(file)
34
35 # Get the results of the compliance check
36 errors = compliance_checker.GetErrors()
37 warnings = compliance_checker.GetWarnings()
38 failed_checks = compliance_checker.GetFailedChecks()
39 return errors, warnings, failed_checks

69 def find_materials(self, stage, find_first:bool=True):
70 '''
71 @brief This function finds the first material in the stage. Assumes MaterialX
72 materials are stored under the "/MaterialX/Materials" scope.
73 @param stage: The stage to search for the first material.
74 @return: The first material found in the stage. If no material is found, None is returned.
75 '''
76 found_materials = []
77
78 # Find the first material under the "MaterialX/Materials" scope
79 materialx_prim = stage.GetPrimAtPath("/MaterialX")
80 if not materialx_prim:
81 print("> Warning: Could not find /MaterialX scope in the USDA file.")
82 return found_materials
83
84 materials_prim = materialx_prim.GetPrimAtPath("Materials")
85 if not materials_prim:
86 print("> Warning: Could not find /MaterialX/Materials scope in the USDA file.")
87 return found_materials
88
89 for child_prim in materials_prim.GetAllChildren():
90 if child_prim.GetTypeName() == "Material":
91 found_materials.append(child_prim)
92 if find_first:
93 break
94
95 return found_materials

97 def add_skydome_light(self,
98 stage,
99 environment_path:str,
100 root_path:str = "/TestScene/Lights",
101 light_name:str = "EnvironmentLight",
102 xform_scale=Gf.Vec3f(1.3, 1.3, 1.3), xform_rotate=Gf.Vec3f(0, 0, 0)):
103 '''
104 @brief This function adds a skydome light to the stage.
105 @param stage: The stage to add the skydome light.
106 @param environment_path: The path to the environment light file.
107 @param root_path: The root path to add the skydome light.
108 @param light_name: The name of the skydome light.
109 @param xform_scale: The scale of the skydome light.
110 @param xform_rotate: The rotation of the skydome light.
111 @return: The skydome light added to the stage.
112 '''
113 skydome_prim = stage.DefinePrim(root_path, "Xform")
114 # Make the skydome prim Xformable
115 xformable = UsdGeom.Xformable(skydome_prim)
116
117 # Scale drawing of skydome
118 scale_op = xformable.GetScaleOp()
119 if not scale_op:
120 scale_op = xformable.AddScaleOp(UsdGeom.XformOp.PrecisionFloat)
121 scale_value = xform_scale
122 scale_op.Set(scale_value)
123
124 dome_light = UsdLux.DomeLight.Define(stage, f"{root_path}/{light_name}")
125
126 # Set the attributes for the DomeLight
127 dome_light.CreateIntensityAttr().Set(1.0)
128 dome_light.CreateTextureFileAttr().Set(environment_path)
129
130 # Set guideRadius
131 dome_light.CreateGuideRadiusAttr().Set(1.0)
132
133 # Rotate the light as needed.
134 xformable = UsdGeom.Xformable(dome_light)
135 xform_op = xformable.GetXformOp(UsdGeom.XformOp.TypeRotateXYZ)
136 if not xform_op:
137 xform_op = xformable.AddXformOp(UsdGeom.XformOp.TypeRotateXYZ, UsdGeom.XformOp.PrecisionFloat)
138 xform_op.Set(xform_rotate)
139
140 # Set the xformOpOrder
141 xformable.SetXformOpOrder([xform_op])
142
143 return dome_light

168 def add_camera(self,
169 stage,
170 camera_path:str,
171 root_path:str="/TestScene/Camera",
172 geometry_path:str="/TestScene/Geometry"):
173 '''
174 @brief This function adds a camera to the stage.
175 @param stage: The stage to add the camera.
176 @param camera_path: The path to the camera file.
177 @param root_path: The root path to add the camera.
178 '''
179 if camera_path:
180 camera = stage.DefinePrim(root_path, "Xform")
181 camera.GetReferences().AddReference(camera_path)
182 return camera
183
184 # Define the geometry path (e.g., a cube or any other geometry)
185 geometry_path = Sdf.Path(geometry_path)
186
187 # Get the UsdPrim for the geometry
188 geometry_prim = stage.GetPrimAtPath(geometry_path)
189
190 camera_path = Sdf.Path(root_path)
191 camera = UsdGeom.Camera.Define(stage, camera_path)
192
193 # Compute the world space bounding box of the geometry
194 bbox_cache = UsdGeom.BBoxCache(Usd.TimeCode.Default(), [UsdGeom.Tokens.default_])
195 bbox = bbox_cache.ComputeWorldBound(geometry_prim)
196 bbox_range = bbox.ComputeAlignedRange()
197
198 # Get the center and size of the bounding box
199 bbox_center = bbox_range.GetMidpoint() # This is a Gf.Vec3d
200 bbox_size = bbox_range.GetSize() # This is a Gf.Vec3d
201
202 # Position the camera to frame the bounding box
203 # Move the camera back along the Z-axis to fit the bounding box
204 distance = max(bbox_size) * 1.5 # Adjust the multiplier as needed
205
206 # Convert bbox_center to Gf.Vec3f for compatibility with Gf.Vec3f(0, 0, distance)
207 camera_position = Gf.Vec3f(bbox_center) + Gf.Vec3f(0, 0, distance)
208 camera.AddTranslateOp().Set(camera_position)
209
210 # Orient the camera to look at the center of the bounding box
211 camera.AddRotateYOp().Set(0) # Rotate to look along the -Z axis
212 look_at = UsdGeom.XformCommonAPI(camera)
213 look_at.SetRotate((0, 0, 0)) # Ensure the camera is aligned
214
215 # Adjust the camera's field of view to fit the bounding box
216 focal_length = 35.0 # Default focal length
217 camera.GetFocalLengthAttr().Set(focal_length)
218
219 # Set the horizontal and vertical aperture to ensure proper framing
220 horizontal_aperture = bbox_size[0] / bbox_size[1] * 20.0 # Adjust based on aspect ratio
221 vertical_aperture = horizontal_aperture
222 camera.GetHorizontalApertureAttr().Set(horizontal_aperture)
223 camera.GetVerticalApertureAttr().Set(vertical_aperture)
224
225 # Set the clipping range to include the bounding box
226 near_clip = distance - max(bbox_size) * 0.5
227 far_clip = distance + max(bbox_size) * 0.5
228 camera.GetClippingRangeAttr().Set(Gf.Vec2f(near_clip, far_clip))
229
230 #camera.SetActive(True)
231
232 # Save the USD stage
233 return camera
