60 def add_materials_for_shaders(doc: mx.Document):
61 '''
62 @brief Add materials for shaders at the root level of a MaterialX document. Nodegraphs are not considered as this is not supported by USD.
63 @param doc The MaterialX document.
64 @return The number of materials added.
65 '''
66 # If has materials skip
67 materials = doc.getMaterialNodes()
68 if len(materials):
69 return 0
70
71 material_count = 0
72
73 surfaceshader_nodes = doc.getChildren()
74 for surfaceshader_node in surfaceshader_nodes:
75 if surfaceshader_node.getCategory() not in ['surfaceshader']:
76 continue
77 downstream_ports = surfaceshader_node.getDownstreamPorts()
78 if not downstream_ports:
79 # Add a material for the shader
80 material_name = doc.createValidChildName('material_' + surfaceshader_node.getName())
81 material_node = doc.addMaterialNode(material_name, surfaceshader_node)
82 if material_node:
83 material_count += 1
84
85 return material_count

88 def add_downstream_materials(doc: mx.Document):
89 '''
90 @brief Add downstream materials to the MaterialX graph.
91 @param doc The MaterialX document.
92 @return The number of materials added.
93 '''
94 # If has materials skip
95 material_count = len(doc.getMaterialNodes())
96 if material_count > 0:
97 return 0
98
99 nodegraphs = doc.getNodeGraphs()
100 if not nodegraphs:
101 return 0
102
103 # Only support these types of graph outputs
104 supported_output_types = [ 'float', 'vector2', 'vector3', 'vector4', 'integer', 'boolean', 'color3', 'color4' ]
105
106 for graph in nodegraphs:
107 if graph.hasSourceUri():
108 continue
109
110 graph_outputs = graph.getOutputs()
111 if not graph_outputs:
112 continue
113
114 # Use does not support these nodes so need to do it the hard way....
115 usd_supports_convert_to_surface_shader = False
116
117 downstream_ports = graph.getDownstreamPorts()
118 if not downstream_ports:
119 # Add a material per output
120 #
121 is_multi_output = len(graph_outputs) > 1
122 for output in graph_outputs:
123 output_name = output.getName()
124 output_type = output.getType()
125
126 if output_type in supported_output_types:
127
128 if usd_supports_convert_to_surface_shader:
129 # Create a new material node
130 shadernode_name = doc.createValidChildName('shader_' + graph.getName() + '_' + output_name)
131 materialnode_name = doc.createValidChildName('material_' + graph.getName() + '_' + output_name)
132
133 convert_definition = 'ND_convert_' + output_type + '_color3'
134 convert_node = doc.getNodeDef(convert_definition)
135 if not convert_node:
136 print("> Failed to find conversion definition: %s" % convert_definition)
137 else:
138 shadernode = doc.addNodeInstance(convert_node, shadernode_name)
139 shadernode.removeAttribute('nodedef')
140 new_input = shadernode.addInput('in', output_type)
141 new_input.setNodeGraphString(graph.getName())
142 new_input.removeAttribute('value')
143 #if is_multi_output:
144 # ISSUE: USD does not handle nodegraph without an explicit output propoerly
145 # so always added in the output string !
146 new_input.setOutputString(output_name)
147 materialnode = doc.addMaterialNode(materialnode_name, shadernode)
148
149 if materialnode:
150 material_count += 1
151
152 else:
153 print(f'Scan: {graph.getName()} output: {output_name} type: {output_type}')
154
155 # If not color3 or float add a convert node and connect it to the current upstream node
156 # and then add in a new output which is of type color3
157 if output_type != 'color3' and output_type != 'float':
158
159 convert_definition = 'ND_convert_' + output_type + '_color3'
160 convert_nodedef = doc.getNodeDef(convert_definition)
161 if not convert_nodedef:
162 print("> Failed to find conversion definition: %s" % convert_definition)
163 continue
164
165 # Find upstream node or interface input
166 convert_upstream = None
167 if len(output.getNodeName()) > 0:
168 convert_upstream = output.getNodeName()
169 elif output.hasInterfaceName():
170 convert_upstream = output.getInterfaceName()
171 if not convert_upstream:
172 print("> Failed to find upstream node for output: %s" % output.getName())
173 continue
174
175 # Insert convert node
176 convert_node = graph.addNodeInstance(convert_nodedef, graph.createValidChildName(f'convert_{convert_upstream}'))
177 convert_node.removeAttribute('nodedef')
178 convert_input = convert_node.addInput('in', output_type)
179 if len(output.getNodeName()) > 0:
180 convert_input.setNodeName(output.getNodeName())
181 elif output.hasInterfaceName():
182 convert_input.setInterfaceName(output.getInterfaceName())
183
184 # Overwrite the upstream connection on the output
185 # and change it's type
186 output.setNodeName(convert_node.getName())
187 output.setType('color3')
188 output_type = 'color3'
189
190 # Create downstream (umlit) shader
191 shadernode_name = doc.createValidChildName('shader_' + graph.getName() + '_' + output_name)
192 materialnode_name = doc.createValidChildName('material_' + graph.getName() + '_' + output_name)
193 unlitDefinition = 'ND_surface_unlit'
194 unlitNode = doc.getNodeDef(unlitDefinition)
195 shadernode = doc.addNodeInstance(unlitNode, shadernode_name)
196 shadernode.removeAttribute('nodedef')
197
198 # Connect upstream output to shader input (based on type)
199 new_input = None
200 if output_type == 'color3':
201 new_input = shadernode.addInput('emission_color', output_type)
202 else:
203 new_input = shadernode.addInput('emission', output_type)
204 new_input.setNodeGraphString(graph.getName())
205 new_input.removeAttribute('value')
206 #if is_multi_output:
207 # ISSUE: USD does not handle nodegraph without an explicit output propoerly
208 # so always added in the output string !
209 new_input.setOutputString(output_name)
210
211 # Add downstream material node connected to shadernode
212 materialnode = doc.addMaterialNode(materialnode_name, shadernode)
213
214 if materialnode:
215 material_count += 1
216
217 return material_count

220 def add_explicit_geometry_stream(graph: mx.GraphElement):
221 '''
222 @brief Add explicit geometry stream nodes for inputs with defaultgeomprop specified
223 in nodes definition. Do this for unconnected inputs only.
224 @param graph The MaterialX graph element.
225 @return The number of implicit nodes added.
226 '''
227
228 graph_default_nodes = {}
229
230 for node in graph.getNodes():
231 if node.hasSourceUri() or (node.getCategory() in ["nodedef"]):
232 continue
233
234 nodedef = node.getNodeDef(node.getType())
235 #print('Node:', node.getName(), 'NodeDef:', nodedef.getName() if nodedef else "None")
236 if not nodedef:
237 continue
238
239 for nodedef_input in nodedef.getInputs():
240 node_input = node.getInput(nodedef_input.getName())
241 # Skip if is a connected input
242 if node_input:
243 if node_input.getInterfaceName() or node_input.getNodeName() or node_input.getNodeGraphString():
244 continue
245
246 # Skip if no defaultgeomprop
247 defaultgeomprop = nodedef_input.getDefaultGeomProp()
248 if not defaultgeomprop:
249 continue
250
251 # Firewall. USD does not appear to handle bitangent properly so
252 # skip it for now.
253 if defaultgeomprop.getGeomProp() == "bitangent":
254 print(f' > WARNING: Skipping adding explicit bitangent node for "{node.getNamePath()}"')
255 continue
256
257 # Fix this up to get information from the defaultgromprop e.g.
258 # - texcoord <geompropdef name="UV0" type="vector2" geomprop="texcoord" index="0">
259 defaultgeomprop_name = defaultgeomprop.getName()
260 defaultgeomprop_prop = defaultgeomprop.getGeomProp()
261 defaultgeomprop_type = defaultgeomprop.getType()
262 defaultgeomprop_index = defaultgeomprop.getIndex()
263 defaultgeomprop_space = defaultgeomprop.getSpace()
264
265 if not node_input:
266 node_input = node.addInput(nodedef_input.getName(), nodedef_input.getType())
267 if defaultgeomprop_name not in graph_default_nodes:
268 upstream_default_node = graph.addNode(defaultgeomprop_prop,
269 graph.createValidChildName(defaultgeomprop_name),
270 defaultgeomprop_type)
271 upstream_default_node.addInputsFromNodeDef()
272
273 # Set space and set index
274 index_input = upstream_default_node.getInput("index")
275 if index_input:
276 index_input.setValue(defaultgeomprop_index, 'integer')
277 space_input = upstream_default_node.getInput("space")
278 if space_input:
279 space_input.setValue(defaultgeomprop_space, 'string')
280
281 #print(f' > Added upstream node "{upstream_default_node.getNamePath()}" : {upstream_default_node}')
282 graph_default_nodes[defaultgeomprop_name] = upstream_default_node
283 else:
284 upstream_default_node = graph_default_nodes[defaultgeomprop_name]
285 #print('Use upstream node for defaultgromprop:', nodedef_input.getName(), defaultgeomprop)
286 node_input.setNodeName(upstream_default_node.getName())
287
288 implicit_nodes_added = len(graph_default_nodes)
289 if graph.getCategory() not in "nodegraph":
290 for child_graph in graph.getNodeGraphs():
291 if child_graph.hasSourceUri():
292 continue
293 implicit_nodes_added += MaterialXUsdUtilities.add_explicit_geometry_stream(child_graph)
294
295 return implicit_nodes_added

298 def encapsulate_top_level_nodes(doc: mx.Document, nodegraph_name:str="top_level_nodes", remove_original:bool=True):
299 """
300 @brief Encapsulate top level nodes in a nodegraph. Remap any connections to the top level nodes
301 to outputs on a new nodegraph.
302 @param doc The MaterialX document.
303 @param nodegraph_name The name of the new nodegraph to encapsulate the top level nodes. Default is 'top_level_nodes'.
304 @param remove_original If True, remove the original top level nodes from the document. Default is True.
305 @return The number of top level nodes found
306 """
307 connections_made = 0
308 top_level_nodes_found = 0
309
310 # Find all children of document which are no material or shader nodes.
311 top_level_nodes = []
312 top_level_connections = []
313 for elem in doc.getNodes():
314 # skips elements that are part of the stdlib
315 if elem.hasSourceUri():
316 continue
317 if (
318 elem.getName()
319 and (elem.getType() not in ["material", "surfaceshader"])
320 and elem.getCategory() not in ["nodegraph", "nodedef"]
321 ):
322 #print("Finding top level nodes: ", elem.getName(), elem.getType())
323 top_level_nodes.append(elem)
324 elif elem.getType() in ["surfaceshader"]:
325 for input_port in elem.getInputs():
326 upstream_node = input_port.getNodeName()
327 if len(upstream_node) > 0:
328 upstream_output_name = input_port.getOutputString()
329 #print("Store connection: ", upstream_node, "<--", input_port.getNamePath())
330 top_level_connections.append([upstream_node, input_port.getNamePath(), upstream_output_name])
331
332 #print("Top level connections: ", top_level_connections)
333 top_level_nodes_found = len(top_level_nodes)
334 if top_level_nodes_found == 0:
335 return top_level_nodes_found
336
337 # create nodegraph
338 ng_name = doc.createValidChildName(nodegraph_name)
339 ng = doc.addNodeGraph(ng_name)
340 for node in top_level_nodes:
341 #print("Adding node: ", node.getName())
342 new_node = ng.addNode(node.getCategory(), mx.createValidName(node.getName()), node.getType())
343 new_node.copyContentFrom(node)
344 for connect in top_level_connections:
345 if connect[0] == node.getName():
346 the_input = doc.getDescendant(connect[1])
347 if not the_input:
348 continue
349
350 # Create a new output on the graph
351 new_output = ng.addOutput(ng.createValidChildName("out"), the_input.getType())
352 new_output.setNodeName(new_node.getName())
353 if len(connect[2]) > 0:
354 new_output.setOutputString(connect[2])
355 #print("Create new output: ", mx.prettyPrint(new_output))
356 the_input.setNodeGraphString(ng_name)
357 the_input.setOutputString(new_output.getName())
358 the_input.removeAttribute("nodename")
359 #print(f"Reconnecting {the_input.getNamePath()} {connect[1]} to {mx.prettyPrint(the_input)}")
360 connections_made += 1
361
362 if remove_original:
363 for node in top_level_nodes:
364 doc.removeChild(node.getName())
365
366 return top_level_nodes_found

369 def encapsulate_top_level_nodes_file(input_path:str, new_input_path:str, nodegraph_name:str='top_level_nodes',
370 remove_original_nodes:bool =True):
371 '''
372 @brief Encapsulate top level nodes in a nodegraph. Remap any connections to the top level nodes
373 to outputs on a new nodegraph.

375 @param new_input_path The path to write the modified MaterialX document.
376 @param nodegraph_name The name of the new nodegraph to encapsulate the top level nodes. Default is 'top_level_nodes'.
377 @param remove_original_nodes If True, remove the original top level nodes from the document. Default is True.

384 if new_input_path:

386 MaterialXUsdUtilities.write_document(doc, new_input_path)
