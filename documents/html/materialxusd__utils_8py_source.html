16 '''
17 @brief Encapsulate top level nodes in a nodegraph. Remap any connections to the top level nodes
18 to outputs on a new nodegraph.
19 @param input_path The path to the MaterialX document.
20 @param new_input_path The path to write the modified MaterialX document.
21 @param nodegraph_name The name of the new nodegraph to encapsulate the top level nodes. Default is 'top_level_nodes'.
22 @param remove_original_nodes If True, remove the original top level nodes from the document. Default is True.
23 @return The modified MaterialX document if top level connections were found, None otherwise.
24 '''
25 connections_made = 0
26
27 doc = mx.createDocument()
28 mx.readFromXmlFile(doc, input_path)
29 # Find all children of document which are no material or shader nodes.
30 top_level_nodes = []
31 top_level_connections = []
32 for elem in doc.getChildren():
33 if elem.getName() and (elem.getType() not in ["material", "surfaceshader"]) and elem.getCategory() not in ["nodegraph"]:
34 #print('Finding top level nodes: ', elem.getName(), elem.getType())
35 top_level_nodes.append(elem)
36 elif elem.getType() in ["surfaceshader"]:
37 for input in elem.getInputs():
38 upstreamNode = input.getNodeName()
39 if len(upstreamNode) > 0:
40 upstreamOutputName = input.getOutputString()
41 #print('Store connection: ', upstreamNode, "<--", input.getNamePath())
42 top_level_connections.append([upstreamNode, input.getNamePath(), upstreamOutputName])
43
44 #print('Top level connections: ', top_level_connections)
45 if len(top_level_connections) > 0:
46 converted_document = True
47
48 ng_name = doc.createValidChildName(nodegraph_name)
49 ng = doc.addNodeGraph(ng_name)
50 for node in top_level_nodes:
51 #print("Adding node: ", node.getName())
52 new_node = ng.addNode(node.getCategory(), mx.createValidName(node.getName()), node.getType())
53 new_node.copyContentFrom(node)
54 for connect in top_level_connections:
55 if connect[0] == node.getName():
56 the_input = doc.getDescendant(connect[1])

62 new_output.setOutputString(connect[2])
