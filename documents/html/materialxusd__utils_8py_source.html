59 graph_default_nodes = {}
60
61 for node in graph.getNodes():
62 if node.hasSourceUri() or (node.getCategory() in ["nodedef"]):
63 continue
64
65 nodedef = node.getNodeDef(node.getType())
66 #print('Node:', node.getName(), 'NodeDef:', nodedef.getName() if nodedef else "None")
67 if not nodedef:
68 continue
69
70 for nodedef_input in nodedef.getInputs():
71 node_input = node.getInput(nodedef_input.getName())
72 # Skip if is a connected input
73 if node_input:
74 if node_input.getInterfaceName() or node_input.getNodeName() or node_input.getNodeGraphString():
75 continue
76
77 # Skip if no defaultgeomprop
78 defaultgeomprop = nodedef_input.getDefaultGeomProp()
79 if not defaultgeomprop:
80 continue
81
82 # Firewall. USD does not appear to handle bitangent properly so

84 if defaultgeomprop.getGeomProp() == "bitangent":
85 print(f' > WARNING: Skipping adding explicit bitangent node for "{node.getNamePath()}"')
86 continue
87
88 # Fix this up to get information from the defaultgromprop e.g.
89 # - texcoord <geompropdef name="UV0" type="vector2" geomprop="texcoord" index="0">
90 defaultgeomprop_name = defaultgeomprop.getName()
91 defaultgeomprop_prop = defaultgeomprop.getGeomProp()
92 defaultgeomprop_type = defaultgeomprop.getType()
93 defaultgeomprop_index = defaultgeomprop.getIndex()
94
95 if not node_input:
96 node_input = node.addInput(nodedef_input.getName(), nodedef_input.getType())
97 if defaultgeomprop_name not in graph_default_nodes:
98 upstream_default_node = graph.addNode(defaultgeomprop_prop,
99 graph.createValidChildName(defaultgeomprop_prop),
100 defaultgeomprop_type)
101 upstream_default_node.addInputsFromNodeDef()
102 index_input = upstream_default_node.getInput("index")
103 if index_input:
104 index_input.setValue(defaultgeomprop_index, 'integer')
105
106 print(f' > Added upstream node "{upstream_default_node.getNamePath()}" : {upstream_default_node}')
107 graph_default_nodes[defaultgeomprop_name] = upstream_default_node
108 else:
109 upstream_default_node = graph_default_nodes[defaultgeomprop_name]
110 #print('Use upstream node for defaultgromprop:', nodedef_input.getName(), defaultgeomprop)
111 node_input.setNodeName(upstream_default_node.getName())

123 def encapsulate_top_level_nodes(doc: mx.Document, nodegraph_name:str="top_level_nodes", remove_original:bool=True):
124 """
125 @brief Encapsulate top level nodes in a nodegraph. Remap any connections to the top level nodes
126 to outputs on a new nodegraph.
127 @param input_path The path to the MaterialX document.
128 @param new_input_path The path to write the modified MaterialX document.
129 @param nodegraph_name The name of the new nodegraph to encapsulate the top level nodes. Default is 'top_level_nodes'.
130 @param remove_original_nodes If True, remove the original top level nodes from the document. Default is True.
131 @return The number of top level nodes found
132 """
133 connections_made = 0
134 top_level_nodes_found = 0
135
136 # Find all children of document which are no material or shader nodes.
137 top_level_nodes = []
138 top_level_connections = []
139 for elem in doc.getNodes():
140 # skips elements that are part of the stdlib
141 if elem.hasSourceUri():
142 continue
143 if (
144 elem.getName()
145 and (elem.getType() not in ["material", "surfaceshader"])
146 and elem.getCategory() not in ["nodegraph", "nodedef"]
147 ):
148 #print("Finding top level nodes: ", elem.getName(), elem.getType())
149 top_level_nodes.append(elem)
150 elif elem.getType() in ["surfaceshader"]:
151 for input_port in elem.getInputs():
152 upstream_node = input_port.getNodeName()
153 if len(upstream_node) > 0:
154 upstream_output_name = input_port.getOutputString()
155 #print("Store connection: ", upstream_node, "<--", input_port.getNamePath())
156 top_level_connections.append([upstream_node, input_port.getNamePath(), upstream_output_name])
157
158 #print("Top level connections: ", top_level_connections)
159 top_level_nodes_found = len(top_level_nodes)
160 if top_level_nodes_found == 0:
161 return top_level_nodes_found
162
163 # create nodegraph
164 ng_name = doc.createValidChildName(nodegraph_name)
165 ng = doc.addNodeGraph(ng_name)
166 for node in top_level_nodes:
167 #print("Adding node: ", node.getName())
168 new_node = ng.addNode(node.getCategory(), mx.createValidName(node.getName()), node.getType())
169 new_node.copyContentFrom(node)
170 for connect in top_level_connections:
171 if connect[0] == node.getName():
172 the_input = doc.getDescendant(connect[1])
173 if not the_input:
174 continue
175
176 # Create a new output on the graph
177 new_output = ng.addOutput(ng.createValidChildName("out"), the_input.getType())
178 new_output.setNodeName(new_node.getName())
179 if len(connect[2]) > 0:
180 new_output.setOutputString(connect[2])
181 #print("Create new output: ", mx.prettyPrint(new_output))
182 the_input.setNodeGraphString(ng_name)
183 the_input.setOutputString(new_output.getName())
184 the_input.removeAttribute("nodename")
185 #print(f"Reconnecting {the_input.getNamePath()} {connect[1]} to {mx.prettyPrint(the_input)}")
186 connections_made += 1
187
188 if remove_original:
189 for node in top_level_nodes:
190 doc.removeChild(node.getName())
191
192 return top_level_nodes_found
